33:/**

35: */
36:public class EditCommand extends Command {
37:
38:    public static final String COMMAND_WORD = "edit";
39:
40:    public static final String MESSAGE_USAGE = COMMAND_WORD + ": Edits the details of the person identified "
41:            + "by the index number used in the displayed person list. "
42:            + "Existing values will be overwritten by the input values.\n"
43:            + "Parameters: INDEX (must be a positive integer) "
44:            + "[" + PREFIX_NAME + "NAME] "
45:            + "[" + PREFIX_PHONE + "PHONE] "
46:            + "[" + PREFIX_EMAIL + "EMAIL] "
47:            + "[" + PREFIX_ADDRESS + "ADDRESS] "
48:            + "[" + PREFIX_EMERGENCY_CONTACT + "EMERGENCY_CONTACT] "
49:            + "[" + PREFIX_TAG + "TAG]...\n"
50:            + "Example: " + COMMAND_WORD + " 1 "
51:            + PREFIX_PHONE + "91234567 "
52:            + PREFIX_EMAIL + "johndoe@example.com";
53:
54:    public static final String MESSAGE_EDIT_PERSON_SUCCESS = "Edited Person: %1$s";
55:    public static final String MESSAGE_NOT_EDITED = "At least one field to edit must be provided.";
56:    public static final String MESSAGE_DUPLICATE_PERSON = "This person already exists in the address book.";
57:
58:    private final Index index;
59:    private final EditPersonDescriptor editPersonDescriptor;
60:
61:    /**
62:     * @param index of the person in the filtered person list to edit
63:     * @param editPersonDescriptor details to edit the person with
64:     */
65:    public EditCommand(Index index, EditPersonDescriptor editPersonDescriptor) {
66:        requireNonNull(index);
67:        requireNonNull(editPersonDescriptor);
68:
69:        this.index = index;
70:        this.editPersonDescriptor = new EditPersonDescriptor(editPersonDescriptor);
71:    }
72:
73:    @Override
74:    public CommandResult execute(Model model) throws CommandException {
75:        requireNonNull(model);
76:        List<Person> lastShownList = model.getFilteredPersonList();
77:
78:        if (index.getZeroBased() >= lastShownList.size()) {
79:            throw new CommandException(Messages.MESSAGE_INVALID_PERSON_DISPLAYED_INDEX);
80:        }
81:
82:        Person personToEdit = lastShownList.get(index.getZeroBased());
83:        Person editedPerson = createEditedPerson(personToEdit, editPersonDescriptor);
84:
85:        if (!personToEdit.isSamePerson(editedPerson) && model.hasPerson(editedPerson)) {
86:            throw new CommandException(MESSAGE_DUPLICATE_PERSON);
87:        }
88:
89:        model.setPerson(personToEdit, editedPerson);
90:        model.updateFilteredPersonList(PREDICATE_SHOW_ALL_PERSONS);
91:        return new CommandResult(String.format(MESSAGE_EDIT_PERSON_SUCCESS, Messages.format(editedPerson)));
92:    }
93:
94:    /**
95:     * Creates and returns a {@code Person} with the details of {@code personToEdit}
96:     * edited with {@code editPersonDescriptor}.
97:     */
98:    private static Person createEditedPerson(Person personToEdit, EditPersonDescriptor editPersonDescriptor) {
99:        assert personToEdit != null;
100:
101:        Name updatedName = editPersonDescriptor.getName().orElse(personToEdit.getName());
102:        Phone updatedPhone = editPersonDescriptor.getPhone().orElse(personToEdit.getPhone());
103:        Email updatedEmail = editPersonDescriptor.getEmail().orElse(personToEdit.getEmail());
104:        Address updatedAddress = editPersonDescriptor.getAddress().orElse(personToEdit.getAddress());
105:        EmergencyContact updatedEmergencyContact =
106:                editPersonDescriptor.getEmergencyContact().orElse(personToEdit.getEmergencyContact());
107:        Set<Tag> updatedTags = editPersonDescriptor.getTags().orElse(personToEdit.getTags());
108:
109:        return new Person(updatedName, updatedPhone, updatedEmail, updatedAddress, updatedEmergencyContact,
110:                updatedTags);
111:    }

114:    public boolean equals(Object other) {
115:        if (other == this) {
116:            return true;
117:        }
118:
119:        // instanceof handles nulls
120:        if (!(other instanceof EditCommand)) {
121:            return false;
122:        }
123:
124:        EditCommand otherEditCommand = (EditCommand) other;

127:    }

135:    }

137:    /**

144:        private Email email;
145:        private Address address;

158:            setEmail(toCopy.email);

160:            setEmergencyContact(toCopy.emergencyContact);
161:            setTags(toCopy.tags);
162:        }
